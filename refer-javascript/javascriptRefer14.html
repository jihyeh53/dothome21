{

");

    for( let i=1; i<100; i++){
        if( i % 2 == 0){
            document.write("<span style= 'color:red'> "+ i +" </span>");
        }else{
            document.write("<span style= 'color:blue'> "+ i +" </span>");

        }
    }
}

07. 테이블 만들기 

{
        document.write("*** 07. 테이블 만들기***
");

        let num = 100;
        let table = "<table>";

        for( let i=1; i<=10; i++){
            table +="<tr>";
            for(let j=1; j<=10; j++ ){
                table +="<td>"+num+"</td>";
                num--;
            }
            table +="</tr>";
        }

        table += "</table>";

        document.write(table);
    }
